library(kableExtra)

# 輸入參數 ----
my_config = fromJSON("./def/my_config.json")
test = FALSE

# run run run ----
unfinished = get_unfinished_eant()
my_url = my_config$ap_url
mail_cc = my_config$mail_cc
email_list = c(unfinished$design_not_yet_list$PIC, unfinished$scoring_not_yet_list$PIC) %>%
        unique() %>%
        sort()
error_id = c()
for(i in email_list){
        
        if(test){
                to_whom = mail_cc
        }else{
                to_whom = i
        }
        
        tryCatch(
                {
                        
                        # STEP0: 輪到哪個 RD/TM ----
                        print(paste0("-------------  ", "user email = ", i, "  -------------"))
                        
                        # STEP1: 切出 user 資料 ----
                        print("STEP1: 切出 user 資料")
                        
                        # show_df = unfinished %>%
                        #         filter(PIC == i)
                        # show_df_html = print(xtable(show_df), 
                        #                      type = "html", 
                        #                      print.results = FALSE)
                        design_temp = unfinished$design_not_yet_list %>%
                                filter(PIC == i)
                        if(nrow(design_temp)>0){
                                show_design_df = design_temp %>%
                                        select(MODEL, PN, ANTENNA_TITLE, PIC) %>% #, ROLE
                                        kbl(align = "c") %>%
                                        kable_classic(full_width = F) %>%
                                        collapse_rows(columns = 1:2, valign = "middle", latex_hline = "full") %>%
                                        row_spec(row = 0:nrow(design_temp), extra_css = "border-bottom: 1px solid")
                        }else{
                                show_design_df = "無"
                        }
                        
                        
                        scoring_temp = unfinished$scoring_not_yet_list %>%
                                filter(PIC == i)
                        if(nrow(scoring_temp)>0){
                                show_scoring_df = scoring_temp %>%
                                        select(MODEL, PN, ANTENNA_TITLE, ROLE, PIC) %>% 
                                        kbl(align = "c",format = "html") %>%
                                        kable_classic(full_width = F) %>%
                                        # column_spec(column = 1:ncol(scoring_temp),
                                        #             border_left = TRUE, #"2px solid gray"
                                        #             border_right = TRUE) %>%
                                        collapse_rows(columns = 1:2, valign = "middle", latex_hline = "full") %>%
                                        row_spec(row = 0:nrow(scoring_temp), extra_css = "border-bottom: 1px solid")
                        }else{
                                show_scoring_df = "無"
                        }
                        
                        
                        # STEP2: 寫信囉 ----
                        print("STEP2: 寫信囉")
                        body_msg = div(
                                p("您好，您所負責的案件已進量產階段，請您至以下", 
                                  tags$a("連結",href = my_url),
                                  "填寫該案天線設計的相關資訊，感謝!"),
                                br(),
                                h5("設計屬性未完成清單"),
                                HTML(as.character(show_design_df)),
                                br(),
                                h5("難度量表未完成清單"),
                                HTML(as.character(show_scoring_df))
                        ) %>% as.character()
                        
                        send.mail(from = "EDSAmail@wnc.com.tw",
                                  to = to_whom,
                                  cc = mail_cc,
                                  subject = "SAS eant Database Notification",
                                  body = body_msg,
                                  html = TRUE,
                                  encoding = "utf-8",
                                  smtp = list(host.name = "twwebmail.wnc.com.tw", 
                                              port = 25,
                                              user.name = "EDSAmail",
                                              passwd = "GvCf7Kz@"), #ssl = TRUE
                                  authenticate = TRUE,
                                  send = TRUE)
                        
                },
                error=function(e){
                        error_id <<- c(error_id, i)
                        print(conditionMessage(e))
                }
        )
        
}

存好後，對檔案點右鍵，把w(write), r(read), x(可執行) 全打勾

用 crontab 

crontab -l #查找目前的排程器(l = list)

30 10 * * * #每天的10:30要進行
30 * * * *  #每小時的30分要進行，例如1:30, 2:30, ..., 23:30, ..

crontab -e #編輯(e = edit)排程器
進去後，按i (i = insert)，可以開始編輯
範例如下：
30 10 * * * /srv/shiny-server_HR/Attendance/batch_mail/trigger.sh

編輯完，打esc，接著按冒號:，然後wq! (write, quit, 強制離開)
